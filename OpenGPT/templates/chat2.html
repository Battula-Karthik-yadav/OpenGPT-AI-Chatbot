function addMessage(text, sender) { const msgDiv = document.createElement("div"); msgDiv.className = `my-2 p-3 rounded max-w-xl whitespace-pre-wrap ${ sender === 'user' ? 'bg-blue-700 self-end text-right' : 'bg-gray-700 self-start text-left' }`; if (sender === 'bot') { msgDiv.innerHTML = marked.parse(text); } else { msgDiv.textContent = text; } chatContainer.appendChild(msgDiv); chatContainer.scrollTop = chatContainer.scrollHeight; return msgDiv; } async function sendMessage() { const text = messageInput.value.trim(); if (!text) return; addMessage(text, "user"); messageInput.value = ""; // Create a placeholder for the bot's response const botMsgDiv = addMessage("", "bot"); try { const response = await fetch("{% url 'send_message' %}", { method: "POST", headers: { "Content-Type": "application/x-www-form-urlencoded", "X-CSRFToken": "{{ csrf_token }}" }, body: new URLSearchParams({ message: text }) }); if (!response.ok) { botMsgDiv.textContent = "Error: Failed to get response."; return; } const reader = response.body.getReader(); const decoder = new TextDecoder("utf-8"); let fullText = ""; while (true) { const { value, done } = await reader.read(); if (done) break; const chunk = decoder.decode(value, { stream: true }); fullText += chunk; botMsgDiv.innerHTML = marked.parse(fullText); chatContainer.scrollTop = chatContainer.scrollHeight; } } catch (err) { botMsgDiv.textContent = "Error contacting server."; console.error(err); } }
